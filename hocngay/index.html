NodeJS

Nền tảng phát triển xây dựng trên Javascript Runtime

Học ngay 
                            
                                var http = require('http');
        
                                http.createServer(function (req, res) {
                                    res.writeHead(200, {'Content-Type': 'text/html'});
                                    res.end('Hello World!');
                                }).listen(8080);
